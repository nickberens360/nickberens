//Toggle extra content
$('.toggle-txt-ex .btn').click(function(){
    $('.additional-content').slideToggle('fast');
});
//Toggle btn text
$(function(){
   $(".btn").click(function () {
      $(this).text(function(i, text){
          return text === "Less" ? "More" : "Less";
      })
   });
})
